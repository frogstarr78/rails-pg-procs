72:         triggers = @connection.triggers(table_name)
73:         unless triggers.empty?
74:           triggers.each {|trigger|
75:             stream.print "  add_trigger(#{Inflector.symbolize(table_name)}" 
76:            events = []
77:             events.push(":insert") if calc(trigger.type, INSERT)
78:             events.push(":update") if calc(trigger.type, UPDATE)
79:             events.push(":delete") if calc(trigger.type, DELETE)
80: 
81:             stream.print ", [" + events.join(", ") + "]"
82:             stream.print ", :before => true" if calc(trigger.type, BEFORE)
83:             stream.print ", :row => true" if calc(trigger.type, ROW)
84:             stream.print ", :name => #{Inflector.symbolize(trigger.name)}" if Inflector.triggerize(table_name, events, calc(trigger.type, BEFORE)) != trigger.name
85:             stream.print ", :function => #{Inflector.symbolize(trigger.procedure_name)}" if Inflector.triggerize(table_name, events, calc(trigger.type, BEFORE)) != trigger.name

88:         end
